const bladesQuotes=[

  "Blades steps up looking focused. He turned pro at 17 — he's been playing pressure golf his whole life.",

  "Blades is dialing in. He once made eagle and six birdies in his first seven holes at the American Express.",

  "Par 5 — Blades goes aggressive. He's always hunting eagles. Always.",

  "Blades takes a deep breath. He played alongside Scottie Scheffler on a Sunday. He doesn't rattle.",

  "Long par 3. Blades loves these — pure iron play, no excuses.",

  "Blades reads the wind. His dad's battle with leukemia taught him composure under pressure.",

  "Another par 5. He fired a final-round 67 at the Byron Nelson to earn PGA Tour membership. He belongs here.",

  "Blades smirks. Everyone knows this is a birdie hole. So does he.",

  "Final hole. Blades is locked in. He said he'd 'go to school on the PGA Tour' at 17. Graduation day."

];

const outcomes={

  safe:[{s:0,w:.55},{s:-1,w:.10},{s:1,w:.30},{s:2,w:.05}],

  standard:[{s:-1,w:.30},{s:0,w:.40},{s:1,w:.22},{s:-2,w:.04},{s:2,w:.04}],

  aggressive:[{s:-2,w:.12},{s:-1,w:.28},{s:0,w:.20},{s:1,w:.25},{s:2,w:.15}]

};

function pick(arr){let r=Math.random(),c=0;for(let i=0;i<arr.length;i++){c+=arr[i].w;if(r<c)return arr[i];}return arr[arr.length-1];}

function bladesShot(idx){

  const hot=idx>=3&&idx<=6;

  const w=hot?[{s:-2,w:.08},{s:-1,w:.45},{s:0,w:.35},{s:1,w:.12}]:[{s:-2,w:.04},{s:-1,w:.35},{s:0,w:.42},{s:1,w:.15},{s:2,w:.04}];

  return pick(w);

}

function scoreName(r){return r<=-2?"Eagle":r===-1?"Birdie":r===0?"Par":r===1?"Bogey":"Double";}

function relStr(n){return n===0?"E":(n>0?"+":"")+n;}

function relColor(n){return n<0?"#0F6E56":n>0?"#A32D2D":"#1a1a1a";}


let state;

function resetGame(){

  state={hole:0,yourTot:0,bladesTot:0,yourScores:[],bladesScores:[],pars:[]};

  document.getElementById("phase-final").classList.add("hidden");

  document.getElementById("phase-result").classList.add("hidden");

  document.getElementById("phase-choose").classList.remove("hidden");

  updateHero(); showHole();

}

function updateHero(){

  const h=state.hole;

  document.getElementById("cur-hole-num").textContent=Math.min(h+1,9);

  document.getElementById("progress-pct").textContent=Math.round(h/9*100)+"% complete";

  document.getElementById("progress").style.width=(h/9*100)+"%";

  const totalPar=state.pars.reduce((a,b)=>a+b,0);

  const ys=totalPar+state.yourTot, bs=totalPar+state.bladesTot;

  const yr=document.getElementById("your-running");

  yr.textContent=relStr(state.yourTot); yr.style.color=relColor(state.yourTot);

  const br=document.getElementById("blades-running");

  br.textContent=relStr(state.bladesTot); br.style.color=relColor(state.bladesTot);

  document.getElementById("your-strokes-total").textContent=h>0?ys+" strokes":"—";

  document.getElementById("blades-strokes-total").textContent=h>0?bs+" strokes":"—";

}

function showHole(){

  const h=holes[state.hole];

  document.getElementById("hole-name").textContent=h.name;

  document.getElementById("hole-meta").textContent="Par "+h.par+" · "+h.yards+" yds";

  document.getElementById("hole-num-badge").textContent=h.num;

  document.getElementById("blades-comment").textContent=bladesQuotes[state.hole];

}

function takeShot(type){

  const h=holes[state.hole];

  const you=pick(outcomes[type]), blades=bladesShot(state.hole);

  const ys=h.par+you.s, bs=h.par+blades.s;

  state.yourTot+=you.s; state.bladesTot+=blades.s;

  state.yourScores.push(you.s); state.bladesScores.push(blades.s); state.pars.push(h.par);

  document.getElementById("phase-choose").classList.add("hidden");

  document.getElementById("phase-result").classList.remove("hidden");

  document.getElementById("res-name").textContent=h.name;

  document.getElementById("res-meta").textContent="Par "+h.par+" · "+h.yards+" yds";

  document.getElementById("res-num").textContent=h.num;

  const yse=document.getElementById("your-strokes");

  yse.textContent=ys; yse.style.color=relColor(you.s);

  const bse=document.getElementById("blades-strokes");

  bse.textContent=bs; bse.style.color=relColor(blades.s);

  const ysn=document.getElementById("your-score-name");

  ysn.textContent=scoreName(you.s); ysn.style.color=relColor(you.s);

  const bsn=document.getElementById("blades-score-name");

  bsn.textContent=scoreName(blades.s); bsn.style.color=relColor(blades.s);

  const diff=you.s-blades.s;

  const ob=document.getElementById("outcome-box");

  ob.className="outcome-box "+(diff<0?"outcome-win":diff>0?"outcome-lose":"outcome-tie");

  document.getElementById("outcome-hl").textContent=diff<0?"You win the hole":diff>0?"Blades takes it":"Halved";

  document.getElementById("outcome-txt").textContent=getCommentary(diff,you.s,blades.s,h,type);

  document.getElementById("next-btn").textContent=state.hole<9?"Next hole →":"See final results →";

  updateHero();

}

function getCommentary(diff,ys,bs,h,type){

  const bn=scoreName(bs).toLowerCase(), yn=scoreName(ys).toLowerCase();

  if(diff<0){

    if(ys<=-2)return "Eagle! That's a statement. Even Blades looks impressed — and he once nearly shot 59 as an 18-year-old.";

    if(ys===-1)return "Birdie! Sharp play. Blades had a "+bn+" — today you outplayed the phenom.";

    return "You take the hole. Blades had a "+bn+". Not his best.";

  }else if(diff>0){

    if(bs<=-2)return "Eagle for Blades. He said he'd go to school on the PGA Tour at 17 — today he's teaching.";

    if(bs===-1)return "Birdie for Blades. Classic. He's been hunting birdies since he was 16.";

    return "Blades edges you. He played with Scottie Scheffler on a Sunday — he doesn't rattle.";

  }else{

    if(ys===0)return "Both par. Solid hole. A clean par is still a par.";

    if(ys===-1)return "Both birdie! The crowd loves it. Great hole.";

    return "Halved — both a "+yn+". Reset and move on.";

  }

}

function nextHole(){

  state.hole++;

  document.getElementById("phase-result").classList.add("hidden");

  if(state.hole>=9){showFinal();}

  else{document.getElementById("phase-choose").classList.remove("hidden");showHole();}

}

function scoreTag(rel,strokes){

  const cls=rel<=-2?"s-eagle":rel===-1?"s-birdie":rel===0?"s-par":rel===1?"s-bogey":"s-double";

  return "<span class='"+cls+"'>"+strokes+"</span>";

}

function showFinal(){

  document.getElementById("phase-final").classList.remove("hidden");

  const totalPar=state.pars.reduce((a,b)=>a+b,0);

  const yt=totalPar+state.yourTot, bt=totalPar+state.bladesTot;
